# ds-core CLI example script
# Lines starting with # are comments

# Check account status
status

# Switch model type (defaults to first entry in config)
model expert

# Simple chat (thinking=false, search=false are optional)
chat Hello, please introduce the Rust language

# Enable thinking mode
chat Explain ownership and borrowing true false

# Enable search
chat What's new in the latest Rust version false true

# Both thinking and search
chat Compare Rust async runtimes true true

# Raw mode: pass through raw SSE bytes without aggregation
chat Hello --raw

# Concurrent test: fire 3 requests at once
concurrent 3 Hello, reply briefly true false

# Concurrent raw mode
concurrent 2 Hello, reply briefly true false --raw

# Exit
quit
